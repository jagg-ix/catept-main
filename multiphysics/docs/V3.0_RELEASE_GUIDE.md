## 💻 **USAGE EXAMPLES**

### Python: Full Implementations

```python
from verification.python.core import registry

# Get equation by label
eq = registry.get_by_label('eq:complex_action')

# Get SymPy expression
expr = eq.sympy_expression()
print(expr)  # I*S_I + S_R

# Verify properties
checks = {
    'positivity': eq.verify_positivity(),
    'hermiticity': eq.verify_hermiticity(),
    'trace': eq.verify_trace(),
    'dimensions': eq.verify_dimensions()
}
print(checks)  # All should be True

# Export to other languages
print(eq.mathematica_code())
print(eq.lean_statement())

# Get all implemented equations
for label, equation in registry.equations.items():
    print(f"{label}: {equation.sympy_expression()}")
```

### Database: Advanced Queries

```sql
-- Overall verification status
SELECT * FROM v_status;

-- Section-by-section progress
SELECT 
  section_name,
  total_equations,
  implemented,
  verified,
  ROUND(100.0 * implemented / total_equations, 1) as percent_complete
FROM v_section_progress
ORDER BY percent_complete DESC;

-- Find specific equation
SELECT 
  id,
  equation_number,
  label,
  section_id,
  status
FROM equations
WHERE label = 'eq:complex_action';

-- Get equation dependencies
SELECT 
  e1.label as equation,
  e2.label as depends_on
FROM dependencies d
JOIN equations e1 ON d.equation_id = e1.id
JOIN equations e2 ON d.depends_on_id = e2.id
WHERE e1.label = 'eq:complex_action';

-- Equations by status
SELECT 
  status,
  COUNT(*) as count
FROM equations
GROUP BY status
ORDER BY count DESC;

-- Recent implementations
SELECT 
  equation_number,
  label,
  implemented_date
FROM equations
WHERE implemented_date IS NOT NULL
ORDER BY implemented_date DESC
LIMIT 10;
```
